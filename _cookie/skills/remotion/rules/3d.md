# 3D Integration

`@remotion/three` integrates Three.js with Remotion for 3D scene rendering. This is optional in COOKIE and used for advanced visual effects.

---

## Setup

```bash
npm install @remotion/three three @types/three @react-three/fiber @react-three/drei
```

---

## ThreeCanvas

The `<ThreeCanvas>` component from `@remotion/three` replaces `<Canvas>` from `@react-three/fiber`, providing Remotion-aware rendering.

```tsx
import { ThreeCanvas } from "@remotion/three";
import { useCurrentFrame, useVideoConfig } from "remotion";

const My3DScene: React.FC = () => {
  const frame = useCurrentFrame();
  const { width, height } = useVideoConfig();

  return (
    <ThreeCanvas
      width={width}
      height={height}
      style={{ backgroundColor: "black" }}
      camera={{ position: [0, 0, 5], fov: 75 }}
    >
      <ambientLight intensity={0.5} />
      <directionalLight position={[5, 5, 5]} intensity={1} />
      <RotatingBox frame={frame} />
    </ThreeCanvas>
  );
};
```

### ThreeCanvas Props

| Prop       | Type       | Description                           |
|------------|------------|---------------------------------------|
| `width`    | `number`   | Canvas width (use from useVideoConfig)|
| `height`   | `number`   | Canvas height                         |
| `camera`   | `object`   | Camera configuration                  |
| `style`    | `CSSProperties` | Container styles                |
| `gl`       | `object`   | WebGL renderer options                |

## Animated 3D Objects

```tsx
import { useFrame } from "@react-three/fiber";
import { useCurrentFrame, useVideoConfig, interpolate, spring } from "remotion";
import { useRef } from "react";
import * as THREE from "three";

const RotatingBox: React.FC<{ frame: number }> = ({ frame }) => {
  const meshRef = useRef<THREE.Mesh>(null);

  const rotationY = interpolate(frame, [0, 150], [0, Math.PI * 2]);
  const rotationX = interpolate(frame, [0, 300], [0, Math.PI]);

  return (
    <mesh ref={meshRef} rotation={[rotationX, rotationY, 0]}>
      <boxGeometry args={[2, 2, 2]} />
      <meshStandardMaterial color="#4F46E5" />
    </mesh>
  );
};
```

## Using useVideoConfig in 3D

```tsx
import { useVideoConfig, useCurrentFrame, interpolate } from "remotion";
import { ThreeCanvas } from "@remotion/three";

const Scene3D: React.FC = () => {
  const { width, height, fps } = useVideoConfig();
  const frame = useCurrentFrame();

  const cameraZ = interpolate(frame, [0, 60], [10, 5], {
    extrapolateRight: "clamp",
  });

  return (
    <ThreeCanvas
      width={width}
      height={height}
      camera={{ position: [0, 0, cameraZ], fov: 60 }}
    >
      <ambientLight intensity={0.4} />
      <pointLight position={[10, 10, 10]} />
      <AnimatedSphere frame={frame} fps={fps} />
    </ThreeCanvas>
  );
};

const AnimatedSphere: React.FC<{ frame: number; fps: number }> = ({ frame, fps }) => {
  const scale = spring({ frame, fps, config: { damping: 12, stiffness: 100 } });

  return (
    <mesh scale={[scale, scale, scale]}>
      <sphereGeometry args={[1, 32, 32]} />
      <meshStandardMaterial color="#10B981" metalness={0.3} roughness={0.4} />
    </mesh>
  );
};
```

## 3D Text with @react-three/drei

```tsx
import { Text3D, Center } from "@react-three/drei";
import { useCurrentFrame, interpolate } from "remotion";

const Animated3DText: React.FC<{ text: string }> = ({ text }) => {
  const frame = useCurrentFrame();
  const opacity = interpolate(frame, [0, 30], [0, 1], {
    extrapolateRight: "clamp",
  });
  const rotY = interpolate(frame, [0, 60], [-0.5, 0], {
    extrapolateRight: "clamp",
  });

  return (
    <Center>
      <Text3D
        font="/fonts/Inter_Bold.json"
        size={1.5}
        height={0.3}
        rotation={[0, rotY, 0]}
      >
        {text}
        <meshStandardMaterial
          color="#FFFFFF"
          transparent
          opacity={opacity}
        />
      </Text3D>
    </Center>
  );
};
```

## Combining 3D with 2D Overlays

```tsx
import { AbsoluteFill, useVideoConfig } from "remotion";
import { ThreeCanvas } from "@remotion/three";

const CompositeScene: React.FC = () => {
  const { width, height } = useVideoConfig();

  return (
    <AbsoluteFill>
      {/* 3D background */}
      <ThreeCanvas width={width} height={height}>
        <ambientLight />
        <Particles />
      </ThreeCanvas>

      {/* 2D overlay on top */}
      <AbsoluteFill
        style={{
          justifyContent: "center",
          alignItems: "center",
        }}
      >
        <h1 style={{ color: "white", fontSize: 80 }}>Title</h1>
      </AbsoluteFill>
    </AbsoluteFill>
  );
};
```

## COOKIE SSD Integration

SSD scenes can configure 3D elements via the `scene_3d` block, mapping to ThreeCanvas and React Three Fiber components.

```yaml
scenes:
  - id: scene-product
    scene_3d:
      enabled: true
      camera:
        position: [0, 0, 5]
        fov: 75
      background: "#000000"
      environment: "studio"        # "studio" | "sunset" | "city" | "forest" | null
      objects:
        - type: "box"             # "box" | "sphere" | "text3d" | "model"
          color: "#4F46E5"
          size: [2, 2, 2]
          animation: "rotate"      # "rotate" | "float" | "scale-in" | null
        - type: "text3d"
          text: "Hello"
          font: "Inter_Bold"
          color: "#FFFFFF"
          animation: "fade-in"
      lighting:
        ambient: 0.5
        directional:
          position: [5, 5, 5]
          intensity: 1.0
```

### Mapping SSD to 3D Components

```tsx
import { ThreeCanvas } from "@remotion/three";
import { useVideoConfig, useCurrentFrame, interpolate, spring } from "remotion";
import { Environment, Text3D, Center } from "@react-three/drei";

const SSD3DScene: React.FC<{ ssdConfig: any }> = ({ ssdConfig }) => {
  const scene3d = ssdConfig.scene_3d;
  if (!scene3d?.enabled) return null;

  const { width, height, fps } = useVideoConfig();
  const frame = useCurrentFrame();

  return (
    <ThreeCanvas
      width={width}
      height={height}
      style={{ backgroundColor: scene3d.background }}
      camera={{
        position: scene3d.camera.position,
        fov: scene3d.camera.fov,
      }}
    >
      <ambientLight intensity={scene3d.lighting.ambient} />
      {scene3d.lighting.directional && (
        <directionalLight
          position={scene3d.lighting.directional.position}
          intensity={scene3d.lighting.directional.intensity}
        />
      )}
      {scene3d.environment && <Environment preset={scene3d.environment} />}
      {scene3d.objects.map((obj: any, i: number) =>
        renderObject(obj, i, frame, fps)
      )}
    </ThreeCanvas>
  );
};

function renderObject(obj: any, key: number, frame: number, fps: number) {
  const scale = obj.animation === "scale-in"
    ? spring({ frame, fps, config: { damping: 12, stiffness: 100 } })
    : 1;
  const rotationY = obj.animation === "rotate"
    ? interpolate(frame, [0, 150], [0, Math.PI * 2])
    : 0;

  switch (obj.type) {
    case "box":
      return (
        <mesh key={key} rotation={[0, rotationY, 0]} scale={[scale, scale, scale]}>
          <boxGeometry args={obj.size} />
          <meshStandardMaterial color={obj.color} />
        </mesh>
      );
    case "sphere":
      return (
        <mesh key={key} scale={[scale, scale, scale]}>
          <sphereGeometry args={[obj.size?.[0] ?? 1, 32, 32]} />
          <meshStandardMaterial color={obj.color} />
        </mesh>
      );
    case "text3d":
      return (
        <Center key={key}>
          <Text3D font={`/fonts/${obj.font}.json`} size={1.5} height={0.3}>
            {obj.text}
            <meshStandardMaterial color={obj.color} />
          </Text3D>
        </Center>
      );
    default:
      return null;
  }
}
```

## Pitfalls

1. **ThreeCanvas, not Canvas.** Use `ThreeCanvas` from `@remotion/three`, not `Canvas` from `@react-three/fiber`. The Remotion version handles frame-accurate rendering for the render pipeline.

2. **useFrame not compatible.** The `useFrame` hook from R3F runs in the animation loop, which conflicts with Remotion's frame-based rendering. Use `useCurrentFrame()` from Remotion instead and pass the frame value to 3D components.

3. **WebGL context limits.** Multiple `ThreeCanvas` components in one composition may exceed browser WebGL context limits (typically 8-16 contexts).

4. **Font JSON for 3D text.** `Text3D` requires a JSON font file generated from a TTF/OTF using facetype.js. This is a different format than WOFF2.

5. **Performance.** 3D rendering is significantly slower than 2D. Expect longer render times. Keep polygon counts reasonable and avoid complex shader effects.
